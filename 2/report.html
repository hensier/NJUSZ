A. Fetch Stone By Terms

考虑石子数量不超过 444 的特殊情形：当前剩余石子为 1,21,21,2 或 333 时，先手获胜；反之后手获胜。可以发现：

  • 当 k=0k=0k=0 时 pk=1p^k=1pk=1，模 444 余 111。
  • 当 k=1k=1k=1 时 pkp^kpk 为质数，模 444 后的余数可以为 1,2,31,2,31,2,3，但不能为 000。

不妨将问题转化为：给定初始石子数量，每次可以取 1∼31 \sim 31∼3 枚石子，求先手必胜还是后手必胜。则：

  • 当石子数量为 444 的倍数时，后手必胜：当先手取 xxx （x mod 4∈{1,2,3}x \bmod 4 \in \{1,2,3\}xmod4∈{1,2,3}）枚石子时，后手只要取 4−x mod 44-x \bmod 44−xmod4 枚，从而保证了当轮到先手时，石子的数量为 444 的倍数，最终一定会依次仅剩下 444 枚、000 枚。因此后手必胜。
  • 当石子数量不为 444 的倍数时，先手必胜：先手可以先取 x mod 4x \bmod 4xmod4 枚石子，则转化为轮到后手时，石子的数量为 444 的倍数，故先手必胜。

可见，答案仅与初始石子数量是否为 444 的倍数有关。

  • 时间复杂度：O(T)\mathcal O(T)O(T)
  • 空间复杂度：O(1)\mathcal O(1)O(1)
#include <bits/stdc++.h>
using namespace std;
int T, n;
int main() {
    read(T);
    while (T--) {
        read(n);
        puts(n % 4 ? "xly wins!" : "cyh wins!");
    }
    return 0;
}

B. Fliptile

考虑枚举所有情形：每个格子最多被操作一次，总情况数为 2MN≤22252^{MN} \le 2^{225}2MN≤2225，不可行。

不妨先枚举第一行的操作方式。第一行操作后，前两行的格子将受到影响。而为了让整个 N×MN \times MN×M 矩阵均为 000，就必然要让第一行均为 000。

在对第一行进行操作后，对于第一行的格子，只有对其正下方的格子进行操作才能让自身格子被翻转。因此，为了确保第一行均为 000，在第一行的操作方式固定后，第二行的操作方式也随之确定。以此类推，第 3∼M−13 \sim M-13∼M−1 行的操作方式也将被确定。而第 MMM 行的操作必须同时让第 M−1M-1M−1 行和第 MMM 行均为 000，需进行判断。

最后记录最小翻转次数即可。

  • 时间复杂度：O(MN⋅2N)≈O(N2⋅2N)\mathcal O(MN \cdot 2^N) \approx \mathcal O( N^2 \cdot 2^N)O(MN⋅2N)≈O(N2⋅2N)
  • 空间复杂度：O(MN)\mathcal O(MN)O(MN)
#include <bits/stdc++.h>
using namespace std;
int M, N, minf = 226, a[15][15], A[15][15], b[15][15], B[15][15];
int main() {
    scanf("%d%d", &M, &N);
    for (int i = 0; i < M; i++) {
        for (int j = 0; j < N; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    for (int i = 0; i < (1 << N); i++) {
        memcpy(A, a, sizeof(a));
        for (int j = 0; j < N; j++) b[0][j] = i >> (N - j - 1) & 1;
        for (int j = 0; j < M; j++) {
            if (j > 0) {
                for (int k = 0; k < N; k++) {
                    A[j][k] ^= b[j-1][k];
                }
            }
            for (int k = 0; k < N; k++) {
                if (k > 0) A[j][k-1] ^= b[j][k];
                A[j][k] ^= b[j][k];
                if (k < N - 1) A[j][k+1] ^= b[j][k];
            }
            if (j < M - 1) memcpy(b[j+1], A[j], sizeof(A[j]));
        }
        bool flag = true;
        for (int j = 0; j < N; j++) {
            if (A[M-1][j]) {
                flag = false;
                break;
            }
        }
        if (flag) {
            int cur = 0;
            for (int j = 0; j < M; j++) {
                for (int k = 0; k < N; k++) {
                    cur += b[j][k];
                }
            }
            if (cur < minf) {
                minf = cur;
                memcpy(B, b, sizeof(b));
            }
        }
    }
    if (minf == 226) printf("IMPOSSIBLE");
    else {
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) printf("%d ", B[i][j]);
            putchar('\n');
        }
    }
    return 0;
}

    reverse(a.begin(), a.end());

        a[i] = x % 10 + '0';
        tmp = x / 10;

    reverse(a.begin(), a.end());

}

D. Jumpy Dumpy!

最优解为位于同一个长为 KKK 的区间内的炸弹数量的最大值。

排序后，我们只需枚举以每个炸弹为区间左 / 右端点。得到对应的右 / 左端点后，通过二分可以获取位于区间内最大 / 小的炸弹。最后按照序号差统计答案即可。

  • 时间复杂度：O(Nlog⁡N)\mathcal O(N \log N)O(NlogN)。
  • 空间复杂度：O(N)\mathcal O(N)O(N)。
#include <bits/stdc++.h>
using namespace std;
int N, K, ans, jd[1001];
int main() {
    scanf("%d%d", &N, &K);
    for (int i = 1; i <= N; i++) scanf("%d", &jd[i]);
    sort(jd + 1, jd + N + 1);
    for (int i = 1; i <= N; i++) {
        int pos = upper_bound(jd + i + 1, jd + N + 1, jd[i] + K) - jd - 1;
        ans = max(ans, pos - i + 1);
    }
    printf("%d", ans);
    return 0;
}

对 aia_iai​ 求和的过程可用树状数组、线段树等维护。

#include <bits/stdc++.h>
using namespace std;
int N, t[200001];
int lowbit(int x) {
    return x & (-x);
}
void add(int x, int c) {
    for (int i = x; i <= N; i += lowbit(i)) t[i] += c;
}
int query(int x) {
    int s = 0;
    for (int i = x; i; i -= lowbit(i)) s += t[i];
    return s;
}
int find(int x) {
    int l = 1, r = N;
    while (l <= r) {
        int mid = (l + r) >> 1;
        if (query(mid) >= x) r = mid - 1;
        else l = mid + 1;
    }
    return l;
}

F. Meeting

我们只需对两种边权分别进行 dfs，就能求出各自所需时间的所有可能值。最后进行比对即可。

  • 时间复杂度：O(M)≈O(N2)\mathcal O(M) \approx \mathcal O(N^2)O(M)≈O(N2)。
  • 空间复杂度：O(N+M)≈O(N2)\mathcal O(N + M) \approx \mathcal O(N^2)O(N+M)≈O(N2)。
#include <bits/stdc++.h>
using namespace std;
int N, M, cnt, head[17];
vector<int> v[2];
struct node {
    int nxt, to, w[2];
} e[121];
void add_edge(int A, int B, int C, int D) {
    e[++cnt] = (node){head[A], B, {C, D}};
    head[A] = cnt;
}
void dfs(int u, int tot, int type) {
    if (u == N) {
        v[type].push_back(tot);
        return;
    }
    for (int i = head[u]; i; i = e[i].nxt) dfs(e[i].to, tot + e[i].w[type], type);
}
int main() {
    scanf("%d%d", &N, &M);
    for (int i = 1, A, B, C, D; i <= M; i++) {
        scanf("%d%d%d%d", &A, &B, &C, &D);
        add_edge(A, B, C, D);
    }
    for (int i = 0; i < 2; i++) dfs(1, 0, i);
    if (v[0].empty()) {
        printf("IMPOSSIBLE");
        return 0;
    }
    for (int i = 0; i < 2; i++) sort(v[i].begin(), v[i].end());
    for (int i = 0; i < v[0].size(); i++) {
        int pos = lower_bound(v[1].begin(), v[1].end(), v[0][i]) - v[1].begin();
        if (v[1][pos] == v[0][i]) {
            printf("%d", v[0][i]);
            return 0;
        }
    }
    printf("IMPOSSIBLE");
    return 0;
}
